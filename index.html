let self = {};

let partner = {};

let total = 0;


// STEP1 렌더

const selfContainer = document.getElementById("selfContainer");


for(let key in data){

  let html = `<div class="category"><b>${key}</b>`;

  data[key].forEach((t,i)=>{

    html += `<button class="btn" onclick="selectSelf('${key}',${i+1},this)">${i+1}. ${t}</button>`;

  });

  html += `</div>`;

  selfContainer.innerHTML += html;

}


function selectSelf(key,val,el){

  self[key]=val;

  el.parentNode.querySelectorAll(".btn").forEach(b=>b.classList.remove("active"));

  el.classList.add("active");

}


// STEP2

function goStep2(){

  total = Object.values(self).reduce((a,b)=>a+b,0);

  document.getElementById("remain").innerText = total;

  document.getElementById("step1").style.display="none";

  document.getElementById("step2").style.display="block";

  renderPartner();

}


function renderPartner(){

  const pc = document.getElementById("partnerContainer");

  pc.innerHTML="";


  for(let key in data){

    let html = `<div class="category"><b>${key}</b>`;

    data[key].forEach((t,i)=>{

      html += `<button class="btn" onclick="selectPartner('${key}',${i+1},this)">${i+1}. ${t}</button>`;

    });

    html += `</div>`;

    pc.innerHTML+=html;

  }

}


function selectPartner(key,val,el){

  partner[key]=val;

  el.parentNode.querySelectorAll(".btn").forEach(b=>b.classList.remove("active"));

  el.classList.add("active");

  updateRemain();

}


function updateRemain(){

  const used = Object.values(partner).reduce((a,b)=>a+b,0);

  const remain = total - used;

  document.getElementById("remain").innerText = remain;


  document.querySelectorAll("#step2 .btn").forEach(btn=>{

    btn.disabled = remain <= 0;

  });

}


// RESULT

function goResult(){

  document.getElementById("step2").style.display="none";

  document.getElementById("resultPage").style.display="block";


  const myTotal = total;

  const pTotal = Object.values(partner).reduce((a,b)=>a+b,0);


  document.getElementById("myTotal").innerText = myTotal;

  document.getElementById("partnerTotal").innerText = pTotal;


  let msg="";

  if(pTotal>myTotal) msg="🔥 기준이 높습니다";

  else if(pTotal===myTotal) msg="👌 균형 잡힘";

  else msg="😎 여유 있음";


  document.getElementById("msg").innerText=msg;


  const bars = document.getElementById("bars");

  for(let key in self){

    const row = document.createElement("div");

    if(partner[key]-self[key]>=2) row.classList.add("danger");


    row.innerHTML=`

    <div>${key}</div>

    <div class="bar">

      <div class="my-bar" style="width:${self[key]*20}%"></div>

      <div class="partner-bar" style="width:${partner[key]*20}%"></div>

    </div>

    `;

    bars.appendChild(row);

  }

}
